# 순열과 조합

### 순열

```js
// nPr
const getPermutation = (arr, r) => {
  if (r === 1) return arr.map((val) => [val]);

  const result = [];
  arr.forEach((fixed, idx, origin) => {
    const rest = [...origin.slice(0, idx), ...origin.slice(idx + 1)];  // 순열
    const permu = getPermutation(rest, r - 1);
    const attached = permu.map((val) => [fixed, ...val]);
    result.push(...attached);
  });

  return result;
};
```



### 조합

```js
// nCr
const getCombination = (arr, r) => {
  if (r === 1) return arr.map((val) => [val]);

  const result = [];
  arr.forEach((fixed, idx, origin) => {
    const rest = origin.slice(idx + 1);  // 조합
    const comb = getCombination(rest, r - 1);
    const attached = comb.map((val) => [fixed, ...val]);
    result.push(...attached);
  });

  return result;
};
```
