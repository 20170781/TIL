### 약수의 갯수 파악

```js
const getDivisorsCount = (num) => {
  let count = 0;
  for (let i = 1; i <= Math.sqrt(num); i++) {
    if (i ** 2 === num) count += 1;
    else if (num % i === 0) count += 2;
  }

  return count;
};
```

### 약수의 합

```js
function solution(n) {
  let answer = 0;

  for (let i = 1; i <= Math.sqrt(n); i++) {
    if (n % i === 0) {
      answer += i;

      if (n !== i ** 2) {
        answer += n / i;
      }
    }
  }

  return answer;
}
```



### 최소공배수 / 최대공약수

```js
function gcd(x, y) {
  while (y) {
    [x, y] = x > y ? [y, x % y] : [x, y % x];
  }
  return x;
}

function lcm(x, y) {
  return (x * y) / gcd(x, y);
}
```
